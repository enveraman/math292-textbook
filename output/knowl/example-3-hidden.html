\begin{align*} A^2 \amp = \bmat0100^2 = \bmat0000, \end{align*}

\begin{align*} \exp(tA) \amp = I_2 + \sum_{k=1}^\infty \frac{A^kt^k}{n!} = \bmat1001 + t\bmat0100 = \bmat1t01. \end{align*}
